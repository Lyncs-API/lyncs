configure_file(__init__.py.in __init__.py)
file(APPEND "${CMAKE_CURRENT_BINARY_DIR}/__init__.py" "# Informations on the external libraries\n")
foreach(lib ${_EXT_LIBS})
  if(ENABLE_${lib})
    file(APPEND "${CMAKE_CURRENT_BINARY_DIR}/__init__.py" "globals()['${lib}'.lower()+'_enabled'] = True\n")
  else()
    file(APPEND "${CMAKE_CURRENT_BINARY_DIR}/__init__.py" "globals()['${lib}'.lower()+'_enabled'] = False\n")
  endif()
  file(APPEND "${CMAKE_CURRENT_BINARY_DIR}/__init__.py" "globals()['${lib}'.lower()+'_path'] = '${${lib}_PATH}'\n")
endforeach(lib)
